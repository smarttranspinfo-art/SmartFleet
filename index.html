function doPost(e) { var sheet = SpreadsheetApp.openById('TU_SPREADSHEET_ID_AQUI') .getSheetByName("Horas"); var d = JSON.parse(e.postData.contents); sheet.appendRow([ d.fecha, d.conductor, d.actividad, d.inicio, d.fin, d.minutos, d.nota, new Date() ]); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({status:"ok"})) .setMimeType(ContentService.MimeType.JSON); }
